052        public static final double NON_NUMERIC = -.99999;
053        public static final String DEFAULT_MONEY_FORMAT = "$###,###,##0.00";
054        public static final String DEFAULT_PERCENT_FORMAT = "##0.0%";
055        public static final String NBSP = " ";

071        public void beforeBody(TableModel model) {
072            logger.debug("XlsView.init()");
073    
074            moneyFormat = model.getPreferences().getPreference(PreferencesConstants.TABLE_EXPORTABLE + "format.money");
075            if (StringUtils.isEmpty(moneyFormat)) {
076                moneyFormat = DEFAULT_MONEY_FORMAT;
077            }
078            percentFormat = model.getPreferences().getPreference(PreferencesConstants.TABLE_EXPORTABLE + "format.percent");
079            if (StringUtils.isEmpty(percentFormat)) {
080                percentFormat = DEFAULT_PERCENT_FORMAT;
081            }
082    
083            encoding = model.getExportHandler().getCurrentExport().getEncoding();
084    
085            wb = new HSSFWorkbook();
086            sheet = wb.createSheet();
087    
088            if (encoding.equalsIgnoreCase("UTF")) {
089                wb.setSheetName(0, "Export Workbook", HSSFWorkbook.ENCODING_UTF_16);
090            } else if (encoding.equalsIgnoreCase("UNICODE")) {
091                wb.setSheetName(0, "Export Workbook", HSSFWorkbook.ENCODING_COMPRESSED_UNICODE);
092            }
093    
094            styles = initStyles(wb);
095            ps = sheet.getPrintSetup();
096    
097            sheet.setAutobreaks(true);
098            ps.setFitHeight((short) 1);
099            ps.setFitWidth((short) 1);
100    
101            createHeader(model);
102        }
103    
104        public void body(TableModel model, Column column) {
105            if (column.isFirstColumn()) {
106                rownum++;
107                cellnum = 0;
108                hssfRow = sheet.createRow(rownum);
109            }
110    
111            String value = ExportViewUtils.parseXLS(column.getCellDisplay());
112    
113            HSSFCell hssfCell = hssfRow.createCell(cellnum);
114    
115            setCellEncoding(hssfCell);
116    
117            if (column.isEscapeAutoFormat()) {
118                writeToCellAsText(hssfCell, value, "");
119            } else {
120                writeToCellFormatted(hssfCell, value, "");
121            }
122            cellnum++;
123        }
124    
125        public Object afterBody(TableModel model) {
126            if (model.getLimit().getTotalRows() != 0) {
127                totals(model);
128            }
129            return wb;
130        }

132        private void createHeader(TableModel model) {
133            rownum = 0;
134            cellnum = 0;
135            HSSFRow row = sheet.createRow(rownum);
136    
137            List columns = model.getColumnHandler().getHeaderColumns();
138            for (Iterator iter = columns.iterator(); iter.hasNext();) {
139                Column column = (Column) iter.next();
140                String title = column.getCellDisplay();
141                HSSFCell hssfCell = row.createCell(cellnum);
142    
143                setCellEncoding(hssfCell);
144    
145                hssfCell.setCellStyle((HSSFCellStyle) styles.get("titleStyle"));
146                hssfCell.setCellType(HSSFCell.CELL_TYPE_STRING);
147                hssfCell.setCellValue(title);
148                int valWidth = (title + "").length() * WIDTH_MULT;
149                sheet.setColumnWidth(hssfCell.getCellNum(), (short) valWidth);
150    
151                cellnum++;
152            }
153        }

155        private void writeToCellAsText(HSSFCell cell, String value, String styleModifier) {
156            // format text
157            if (value.trim().equals(NBSP)) {
158                value = "";
159            }
160            cell.setCellStyle((HSSFCellStyle) styles.get("textStyle" + styleModifier));
161            fixWidthAndPopulate(cell, NON_NUMERIC, value);
162        }
163    
164        private void writeToCellFormatted(HSSFCell cell, String value, String styleModifier) {
165            double numeric = NON_NUMERIC;
166    
167            try {
168                numeric = Double.parseDouble(value);
169            } catch (Exception e) {
170                numeric = NON_NUMERIC;
171            }
172    
173            if (value.startsWith("$") || value.endsWith("%") || value.startsWith("($")) {
174                boolean moneyFlag = (value.startsWith("$") || value.startsWith("($"));
175                boolean percentFlag = value.endsWith("%");
176    
177                value = StringUtils.replace(value, "$", "");
178                value = StringUtils.replace(value, "%", "");
179                value = StringUtils.replace(value, ",", "");
180                value = StringUtils.replace(value, "(", "-");
181                value = StringUtils.replace(value, ")", "");
182    
183                try {
184                    numeric = Double.parseDouble(value);
185                } catch (Exception e) {
186                    numeric = NON_NUMERIC;
187                }
188    
189                cell.setCellType(HSSFCell.CELL_TYPE_NUMERIC);
190    
191                if (moneyFlag) {
192                    // format money
193                    cell.setCellStyle((HSSFCellStyle) styles.get("moneyStyle" + styleModifier));
194                } else if (percentFlag) {
195                    // format percent
196                    numeric = numeric / 100;
197                    cell.setCellStyle((HSSFCellStyle) styles.get("percentStyle" + styleModifier));
198                }
199            } else if (numeric != NON_NUMERIC) {
200                // format numeric
201                cell.setCellStyle((HSSFCellStyle) styles.get("numericStyle" + styleModifier));
202            } else {
203                // format text
204                if (value.trim().equals(NBSP)) {
205                    value = "";
206                }
207                cell.setCellStyle((HSSFCellStyle) styles.get("textStyle" + styleModifier));
208            }
209    
210            fixWidthAndPopulate(cell, numeric, value);
211        }
212    
213        private void fixWidthAndPopulate(HSSFCell cell, double numeric, String value) {
214            int valWidth = 0;
215    
216            if (numeric != NON_NUMERIC) {
217                cell.setCellValue(numeric);
218                valWidth = (cell.getNumericCellValue() + "$,.").length() * WIDTH_MULT;
219            } else {
220                cell.setCellValue(value);
221                valWidth = (cell.getStringCellValue() + "").length() * WIDTH_MULT;
222    
223                if (valWidth < (WIDTH_MULT * MIN_CHARS)) {
224                    valWidth = WIDTH_MULT * MIN_CHARS;
225                }
226            }
227    
228            if (valWidth > sheet.getColumnWidth(cell.getCellNum())) {
229                sheet.setColumnWidth(cell.getCellNum(), (short) valWidth);
230            }
231        }

399        // Add to export totals
400        public void totals(TableModel model) {
401            Column firstCalcColumn = model.getColumnHandler().getFirstCalcColumn();
402            if (firstCalcColumn != null) {
403                int rows = firstCalcColumn.getCalc().length;
404                for (int i = 0; i < rows; i++) {
405                    rownum++;
406                    HSSFRow row = sheet.createRow(rownum);
407                    cellnum = 0;
408                    for (Iterator iter = model.getColumnHandler().getColumns().iterator(); iter.hasNext();) {
409                        Column column = (Column) iter.next();
410                        if (column.isFirstColumn()) {
411                            String calcTitle = CalcUtils.getFirstCalcColumnTitleByPosition(model, i);
412                            HSSFCell cell = row.createCell(cellnum);
413                            setCellEncoding(cell);
414                            if (column.isEscapeAutoFormat()) {
415                                writeToCellAsText(cell, calcTitle, "_Totals");
416                            } else {
417                                writeToCellFormatted(cell, calcTitle, "_Totals");
418                            }
419                            cellnum++;
420                            continue;
421                        }
422    
423                        if (column.isCalculated()) {
424                            CalcResult calcResult = CalcUtils.getCalcResultsByPosition(model, column, i);
425                            Number value = calcResult.getValue();
426                            HSSFCell cell = row.createCell(cellnum);
427                            setCellEncoding(cell);
428                            if (value != null)
429                                if (column.isEscapeAutoFormat()) {
430                                    writeToCellAsText(cell, value.toString(), "_Totals");
431                                } else {
432                                    writeToCellFormatted(cell, ExtremeUtils.formatNumber(column.getFormat(), value, model.getLocale()), "_Totals");
433                                }
434                            else {
435                                cell.setCellStyle((HSSFCellStyle) styles.get("naStyle_Totals"));
436                                cell.setCellValue("n/a");
437                            }
438                            cellnum++;
439                        } else {
440                            HSSFCell cell = row.createCell(cellnum);
441                            setCellEncoding(cell);
442                            writeToCellFormatted(cell, "", "_Totals");
443                            cellnum++;
444                        }
445                    }
446                }
447            }
448    
449        }
