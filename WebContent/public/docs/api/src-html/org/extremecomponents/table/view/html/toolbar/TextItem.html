016    package org.extremecomponents.table.view.html.toolbar;
017    
018    import org.extremecomponents.table.core.TableModel;
019    import org.extremecomponents.util.HtmlBuilder;
020    
021    /**
022     * @author Jeff Johnston
023     */
024    public class TextItem extends AbstractItem implements ToolbarItem {
025        private String text;
026    
027        public String getText() {
028            return text;
029        }
030    
031        public void setText(String text) {
032            this.text = text;
033        }
034    
035        public void disabled(HtmlBuilder html) {
036            html.span().close().append(getText()).spanEnd();
037        }
038    
039        public void enabled(HtmlBuilder html, TableModel model) {
040            html.a();
041            html.quote();
042            html.append(getAction());
043            html.quote().close();
044    
045            boolean showTooltips = model.getTableHandler().getTable().isShowTooltips();
046            if (showTooltips) {
047                html.span().title(getTooltip());
048                html.styleClass(getStyleClass()).style(getStyle());
049                html.onmouseover(getOnmouseover()).onmouseout(getOnmouseout());
050                html.close();
051                html.append(getText());
052                html.spanEnd();
053            } else {
054                html.span();
055                html.styleClass(getStyleClass()).style(getStyle());
056                html.onmouseover(getOnmouseover()).onmouseout(getOnmouseout());
057                html.close();
058                html.append(getText());
059                html.spanEnd();
060            }
061    
062            html.aEnd();
063        }
064    }
