016    package org.extremecomponents.table.limit;
017    
018    import org.apache.commons.lang.builder.ToStringBuilder;
019    
020    /**
021     * @author Jeff Johnston
022     */
023    public final class TableLimit implements Limit {
024        protected LimitFactory limitFactory;
025        protected FilterSet filterSet;
026        protected Sort sort;
027        protected boolean exported;
028        protected int rowStart;
029        protected int rowEnd;
030        protected int currentRowsDisplayed;
031        protected int page;
032        protected int totalRows;
033    
034        public TableLimit(LimitFactory limitFactory) {
035            this.limitFactory = limitFactory;
036            this.filterSet = limitFactory.getFilterSet();
037            this.sort = limitFactory.getSort();
038            this.page = limitFactory.getPage();
039            this.exported = limitFactory.isExported();
040        }
041    
042        public FilterSet getFilterSet() {
043            return filterSet;
044        }
045    
046        public int getRowEnd() {
047            return rowEnd;
048        }
049    
050        public int getRowStart() {
051            return rowStart;
052        }
053    
054        public Sort getSort() {
055            return sort;
056        }
057    
058        public int getPage() {
059            return page;
060        }
061    
062        public int getCurrentRowsDisplayed() {
063            return currentRowsDisplayed;
064        }
065    
066        public int getTotalRows() {
067            return totalRows;
068        }
069    
070        public boolean isFiltered() {
071            return filterSet.isFiltered();
072        }
073    
074        public boolean isCleared() {
075            return filterSet.isCleared();
076        }
077    
078        public boolean isSorted() {
079            return sort.isSorted();
080        }
081    
082        public boolean isExported() {
083            return exported;
084        }
085    
086        /**
087         * When using the Limit you must call this method to set the following row
088         * attributes: rowStart, rowEnd, currentRowsDisplayed and page
089         * 
090         * @param totalRows The total results for the table.
091         * @param rowsDisplayed The default rows displayed on the table. This
092         *            value is either defined on the table, or defaulted in the
093         *            Preferences.
094         */
095        public void setRowAttributes(int totalRows, int rowsDisplayed) {
096            int currentRowsDisplayed = limitFactory.getCurrentRowsDisplayed(totalRows, rowsDisplayed);
097    
098            int page = getValidPage(this.page, totalRows, currentRowsDisplayed);
099    
100            int rowStart = (page - 1) * currentRowsDisplayed;
101    
102            int rowEnd = rowStart + currentRowsDisplayed;
103    
104            if (rowEnd > totalRows) {
105                rowEnd = totalRows;
106            }
107    
108            this.page = page;
109            this.currentRowsDisplayed = currentRowsDisplayed;
110            this.totalRows = totalRows;
111            this.rowStart = rowStart;
112            this.rowEnd = rowEnd;
113        }
114    
115        /**
116         * The page returned that is not greater than the pages that can display.
117         */
118        private int getValidPage(int page, int totalRows, int currentRowsDisplayed) {
119            if (!isValidPage(page, totalRows, currentRowsDisplayed)) {
120                return getValidPage(--page, totalRows, currentRowsDisplayed);
121            }
122    
123            return page;
124        }
125    
126        /**
127         * Testing that the page returned is not greater than the pages that are
128         * able to be displayed. The problem arises if using the state feature and
129         * rows are deleted.
130         */
131        private boolean isValidPage(int page, int totalRows, int currentRowsDisplayed) {
132            if (page == 1) {
133                return true;
134            }
135            
136            int rowStart = (page - 1) * currentRowsDisplayed;
137            int rowEnd = rowStart + currentRowsDisplayed;
138            if (rowEnd > totalRows) {
139                rowEnd = totalRows;
140            }
141            return rowEnd > rowStart;
142        }
143    
144        public String toString() {
145            ToStringBuilder builder = new ToStringBuilder(this);
146            builder.append("rowStart", rowStart);
147            builder.append("rowEnd", rowEnd);
148            builder.append("currentRowsDisplayed", currentRowsDisplayed);
149            builder.append("page", page);
150            builder.append("totalRows", totalRows);
151            builder.append("exported", exported);
152            builder.append("sort", sort);
153            builder.append("filterSet", filterSet);
154            return builder.toString();
155        }
156    }
