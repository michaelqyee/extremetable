016    package org.extremecomponents.util;
017    
018    import java.io.File;
019    import java.util.Properties;
020    
021    import org.apache.commons.lang.StringUtils;
022    
023    /**
024     * @author phorn
025     */
026    public class MimeUtils {
027        public final static String MIMETYPES_PROPERTIES = "mimeTypes.properties";
028        private static Properties properties;
029    
030        /**
031         * Return the mime type for a file. The extension is parsed out and the
032         * method to get mime type by extension is called. Extension mappings are
033         * found in the mimeTypes.properties file in the org.extremecomponents.util
034         * package.
035         */
036        public static String getFileMimeType(File file) {
037            if (file == null) {
038                return null;
039            }
040    
041            return getFileMimeType(file.getName());
042        }
043    
044        /**
045         * Return the mime type for a file name. The extension is parsed out and the
046         * method to get mime type by extension is called. Extension mappings are
047         * found in the mimeTypes.properties file in the org.extremecomponents.util
048         * package.
049         */
050        public static String getFileMimeType(String fileName) {
051            if (StringUtils.isBlank(fileName) || (fileName.indexOf(".") == -1)) {
052                return null;
053            }
054    
055            fileName = fileName.substring(fileName.lastIndexOf("."));
056    
057            return getExtensionMimeType(fileName);
058        }
059    
060        /**
061         * Return the mime type for a given extension. Extensions can include the
062         * dot or just be the extension without the dot. Extension mappings are
063         * found in the mimeTypes.properties file in the org.extremecomponents.util
064         * package.
065         */
066        public static String getExtensionMimeType(String extension) {
067            String result = null;
068    
069            if (StringUtils.isBlank(extension)) {
070                return result;
071            }
072    
073            init();
074    
075            extension = extension.toLowerCase();
076    
077            if (!extension.startsWith(".")) {
078                extension = "." + extension;
079            }
080    
081            result = (String) properties.get(extension);
082    
083            return result;
084        }
085    
086        private static void init() {
087            if (properties != null) {
088                return;
089            }
090    
091            try {
092                properties = new Properties();
093                properties.load(new MimeUtils().getClass().getResourceAsStream(MIMETYPES_PROPERTIES));
094            } catch (Exception e) {
095                e.printStackTrace();
096            }
097        }
098    
099        /**
100         * Just a test harness.
101         */
102        public static void main(String[] args) {
103            System.out.println("MimeUtils.getExtensionMimeType(.gif)=" + MimeUtils.getExtensionMimeType(".gif"));
104            System.out.println("MimeUtils.getExtensionMimeType(.pdf)=" + MimeUtils.getExtensionMimeType(".pdf"));
105            System.out.println("MimeUtils.getExtensionMimeType(.xls)=" + MimeUtils.getExtensionMimeType(".xls"));
106            System.out.println("MimeUtils.getFileMimeType(foo.gif)=" + MimeUtils.getFileMimeType("foo.gif"));
107            System.out.println("MimeUtils.getFileMimeType(foo.pdf)=" + MimeUtils.getFileMimeType("foo.pdf"));
108            System.out.println("MimeUtils.getFileMimeType(foo.xls)=" + MimeUtils.getFileMimeType("foo.xls"));
109            System.out.println("MimeUtils.getFileMimeType(foo.badextension)=" + MimeUtils.getFileMimeType("foo.badextension"));
110        }
111    }
