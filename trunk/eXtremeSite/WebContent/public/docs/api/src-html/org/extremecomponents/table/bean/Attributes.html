016    package org.extremecomponents.table.bean;
017    
018    import java.util.HashMap;
019    import java.util.Map;
020    
021    /*
022     * @author Jeff Johnston
023     */
024    public abstract class Attributes {
025        private Map attr = new HashMap();
026    
027        public Object getAttribute(String key) {
028            return attr.get(key);
029        }
030    
031        public String getAttributeAsString(String key) {
032            Object value = attr.get(key);
033            if (value != null) {
034                return String.valueOf(value);
035            }
036    
037            return null;
038        }
039    
040        public int getAttributeAsInt(String key) {
041            Object value = attr.get(key);
042            if (value != null) {
043                return Integer.parseInt(String.valueOf(value));
044            }
045    
046            return 0;
047        }
048    
049        public void addAttribute(String key, Object value) {
050            attr.put(key, value);
051        }
052    }
