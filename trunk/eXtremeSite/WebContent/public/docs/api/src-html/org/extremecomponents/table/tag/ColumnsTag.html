001    package org.extremecomponents.table.tag;
002    
003    import javax.servlet.jsp.JspException;
004    import javax.servlet.jsp.tagext.TagSupport;
005    
006    import org.apache.commons.logging.Log;
007    import org.apache.commons.logging.LogFactory;
008    import org.extremecomponents.table.core.TableModel;
009    import org.extremecomponents.util.ExceptionUtils;
010    
011    /**
012     * @jsp.tag name="columns" display-name="ColumnsTag" body-content="JSP"
013     *          description="Auto generate the columns."
014     * 
015     * @author Jeff Johnston
016     */
017    
018    public class ColumnsTag extends TagSupport {
019        private static Log logger = LogFactory.getLog(ColumnsTag.class);
020    
021        private String autoGenerateColumns;
022    
023        /**
024         * @jsp.attribute description="A fully qualified class name to a custom
025         *                AutoGenerateColumns implementation. Could also be a named
026         *                type in the preferences. Used to generate columns on the
027         *                fly." required="true" rtexprvalue="true"
028         */
029        public void setAutoGenerateColumns(String autoGenerateColumns) {
030            this.autoGenerateColumns = autoGenerateColumns;
031        }
032    
033        public int doEndTag() throws JspException {
034            try {
035                TableModel model = TagUtils.getModel(this);
036    
037                if (!TagUtils.isIteratingBody(this)) {
038                    String autoGenerateColumns = TagUtils.evaluateExpressionAsString("autoGenerateColumns", this.autoGenerateColumns, this, pageContext);
039                    model.addColumns(autoGenerateColumns);
040                } else {
041                    model.setColumnValues();
042                }
043    
044                return EVAL_PAGE;
045            } catch (Exception e) {
046                logger.error(ExceptionUtils.formatStackTrace(e));
047                throw new JspException("ColumnsTag.doEndTag() Problem: " + ExceptionUtils.formatStackTrace(e));
048            }
049        }
050    
051        public void release() {
052            autoGenerateColumns = null;
053            super.release();
054        }
055    }
