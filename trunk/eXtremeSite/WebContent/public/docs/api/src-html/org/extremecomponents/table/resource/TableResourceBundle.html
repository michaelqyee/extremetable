016    package org.extremecomponents.table.resource;
017    
018    import java.text.MessageFormat;
019    import java.util.Locale;
020    import java.util.MissingResourceException;
021    import java.util.ResourceBundle;
022    
023    import org.apache.commons.lang.StringUtils;
024    import org.apache.commons.logging.Log;
025    import org.apache.commons.logging.LogFactory;
026    import org.extremecomponents.table.context.Context;
027    import org.extremecomponents.table.core.Messages;
028    import org.extremecomponents.table.core.TableModelUtils;
029    
030    /**
031     * Load up the resource bundle that will be used by the Model.
032     * 
033     * @author Jeff Johnston
034     */
035    public class TableResourceBundle implements Messages {
036        private static Log logger = LogFactory.getLog(TableResourceBundle.class);
037    
038        public final static String EXTREMETABLE_RESOURCE_BUNDLE = "org.extremecomponents.table.resource.extremetableResourceBundle";
039    
040        private ResourceBundle customResourceBundle;
041        private ResourceBundle defaultResourceBundle;
042        private Locale locale;
043    
044        public void init(Context context, Locale locale) {
045            this.locale = locale;
046            defaultResourceBundle = findResourceBundle(EXTREMETABLE_RESOURCE_BUNDLE, locale);
047            String messagesLocation = TableModelUtils.getMessagesLocation(context);
048            if (StringUtils.isNotBlank(messagesLocation)) {
049                customResourceBundle = findResourceBundle(messagesLocation, locale);
050            }
051        }
052    
053        private ResourceBundle findResourceBundle(String resourceBundleLocation, Locale locale) {
054            try {
055                return ResourceBundle.getBundle(resourceBundleLocation, locale, getClass().getClassLoader());
056            } catch (MissingResourceException e) {
057                if (logger.isErrorEnabled()) {
058                    logger.error("The resource bundle [ " + resourceBundleLocation + "] was not found. Make sure the path and resource name is correct.", e);
059                }
060            }
061    
062            return null;
063        }
064    
065        /**
066         * Get the resource property.
067         */
068        public String getMessage(String code) {
069            return getMessage(code, null);
070        }
071    
072        /**
073         * Get the resource property.
074         */
075        public String getMessage(String code, Object[] args) {
076            String result = findResource(customResourceBundle, code);
077    
078            if (result == null) {
079                result = findResource(defaultResourceBundle, code);
080            }
081    
082            if (result != null && args != null) {
083                MessageFormat formatter = new MessageFormat("");
084                formatter.setLocale(locale);
085                formatter.applyPattern(result);
086                result = formatter.format(args);
087            }
088    
089            return result;
090        }
091    
092        private String findResource(ResourceBundle resourceBundle, String code) {
093            String result = null;
094    
095            if (resourceBundle == null) {
096                return result;
097            }
098    
099            try {
100                result = resourceBundle.getString(code);
101            } catch (MissingResourceException e) {
102                // nothing we can do so eat the message
103            }
104    
105            return result;
106        }
107    }
