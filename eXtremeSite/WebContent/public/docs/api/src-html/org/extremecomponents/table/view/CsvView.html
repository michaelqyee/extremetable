016    package org.extremecomponents.table.view;
017    
018    import org.apache.commons.lang.StringUtils;
019    import org.extremecomponents.table.bean.Column;
020    import org.extremecomponents.table.bean.Export;
021    import org.extremecomponents.table.core.TableModel;
022    
023    /**
024     * For producing a delimited datafile view of the table. Default delimiter is
025     * comma.
026     * 
027     * @author Brad Parks
028     */
029    public class CsvView implements View {
030        public final static String DELIMITER = "delimiter";
031        final static String DEFAULT_DELIMITER = ",";
032        private StringBuffer plainData = new StringBuffer();
033    
034        public void beforeBody(TableModel model) {
035        }
036    
037        public void body(TableModel model, Column column) {
038            Export export = model.getExportHandler().getCurrentExport();
039            String delimiter = export.getAttributeAsString(DELIMITER);
040            if (StringUtils.isBlank(delimiter)) {
041                delimiter = DEFAULT_DELIMITER;
042            }
043    
044            String value = ExportViewUtils.parseCSV(column.getCellDisplay());
045    
046            plainData.append(value.trim());
047            plainData.append(delimiter);
048            if (column.isLastColumn()) {
049                plainData.append("\r\n");
050            }
051        }
052    
053        public Object afterBody(TableModel model) {
054            return plainData.toString();
055        }
056    }
