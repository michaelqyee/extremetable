016    package org.extremecomponents.table.tag;
017    
018    import org.apache.commons.lang.StringUtils;
019    import org.extremecomponents.table.bean.Export;
020    import org.extremecomponents.table.core.TableConstants;
021    import org.extremecomponents.table.core.TableModel;
022    import org.extremecomponents.table.view.PdfView;
023    import org.extremecomponents.table.view.html.BuilderConstants;
024    
025    /**
026     * @jsp.tag name="exportPdf" display-name="ExportPdfTag" body-content="JSP"
027     *          description="Export data for a pdf view."
028     * 
029     * @author Jeff Johnston
030     */
031    public class ExportPdfTag extends ExportTag {
032        private String headerBackgroundColor;
033        private String headerTitle;
034        private String headerColor;
035    
036        /**
037         * @jsp.attribute description="The background color on the header column."
038         *                required="false" rtexprvalue="true"
039         */
040        public void setHeaderBackgroundColor(String headerBackgroundColor) {
041            this.headerBackgroundColor = headerBackgroundColor;
042        }
043    
044        /**
045         * @jsp.attribute description="The font color for the header column."
046         *                required="false" rtexprvalue="true"
047         */
048        public void setHeaderColor(String headerColor) {
049            this.headerColor = headerColor;
050        }
051    
052        /**
053         * @jsp.attribute description="The title displayed at the top of the page."
054         *                required="false" rtexprvalue="true"
055         */
056        public void setHeaderTitle(String headerTitle) {
057            this.headerTitle = headerTitle;
058        }
059    
060        public void addExportAttributes(TableModel model, Export export) {
061            if (StringUtils.isBlank(export.getView())) {
062                export.setView(TableConstants.VIEW_PDF);
063            }
064    
065            if (StringUtils.isBlank(export.getViewResolver())){
066                export.setViewResolver(TableConstants.VIEW_PDF);
067            }
068    
069            if (StringUtils.isBlank(export.getImageName())) {
070                export.setImageName(TableConstants.VIEW_PDF);
071            }
072            
073            if (StringUtils.isBlank(export.getText())) {
074                export.setText(BuilderConstants.TOOLBAR_PDF_TEXT);
075            }
076    
077            export.addAttribute(PdfView.HEADER_BACKGROUND_COLOR, TagUtils.evaluateExpressionAsString("headerBackgroundColor", headerBackgroundColor, this, pageContext));
078            export.addAttribute(PdfView.HEADER_COLOR, TagUtils.evaluateExpressionAsString("headerColor", headerColor, this, pageContext));
079            export.addAttribute(PdfView.HEADER_TITLE, TagUtils.evaluateExpressionAsString("headerTitle", headerTitle, this, pageContext));
080        }
081    
082        public void release() {
083            headerBackgroundColor = null;
084            headerTitle = null;
085            headerColor = null;
086            super.release();
087        }
088    }
