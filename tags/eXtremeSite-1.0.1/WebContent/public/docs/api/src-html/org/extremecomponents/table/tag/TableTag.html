393        public int doStartTag() throws JspException {
394            try {
395                // initialize the attributes
396                iterator = null;
397                pageContext.setAttribute(TableConstants.ROWCOUNT, "0");
398    
399                // fire up the model with the context, preferences and messages
400                model = new TableModelImpl(new JspPageContext(pageContext), TagUtils.evaluateExpressionAsString("locale", this.locale, this, pageContext));
401    
402                // make the table
403                Table table = new Table(model);
404                table.setAction(TagUtils.evaluateExpressionAsString("action", action, this, pageContext));
405                table.setAutoIncludeParameters(TagUtils.evaluateExpressionAsBoolean("autoIncludeParameters", this.autoIncludeParameters, this, pageContext));
406                table.setBorder(TagUtils.evaluateExpressionAsString("border", this.border, this, pageContext));
407                table.setBufferView(TagUtils.evaluateExpressionAsBoolean("bufferView", this.bufferView, this, pageContext));
408                table.setCellpadding(TagUtils.evaluateExpressionAsString("cellpadding", this.cellpadding, this, pageContext));
409                table.setCellspacing(TagUtils.evaluateExpressionAsString("cellspacing", this.cellspacing, this, pageContext));
410                table.setFilterable(TagUtils.evaluateExpressionAsBoolean("filterable", this.filterable, this, pageContext));
411                table.setFilterRowsCallback(TagUtils.evaluateExpressionAsString("filterRowsCallback", this.filterRowsCallback, this, pageContext));
412                table.setForm(TagUtils.evaluateExpressionAsString("form", this.form, this, pageContext));
413                table.setImagePath(TagUtils.evaluateExpressionAsString("imagePath", this.imagePath, this, pageContext));
414                table.setInterceptor(TagUtils.evaluateExpressionAsString("interceptor", this.interceptor, this, pageContext));
415                table.setItems(TagUtils.evaluateExpressionAsObject("items", this.items, this, pageContext));
416                table.setLocale(TagUtils.evaluateExpressionAsString("locale", this.locale, this, pageContext));
417                table.setMethod(TagUtils.evaluateExpressionAsString("method", this.method, this, pageContext));
418                table.setOnInvokeAction(TagUtils.evaluateExpressionAsString("onInvokeAction", this.onInvokeAction, this, pageContext));
419                table.setRetrieveRowsCallback(TagUtils.evaluateExpressionAsString("retrieveRowsCallback", this.retrieveRowsCallback, this, pageContext));
420                table.setRowsDisplayed(TagUtils.evaluateExpressionAsInt("rowsDisplayed", this.rowsDisplayed, this, pageContext));
421                table.setScope(TagUtils.evaluateExpressionAsString("scope", scope, this, pageContext));
422                table.setShowExports(TagUtils.evaluateExpressionAsBoolean("showExports", this.showExports, this, pageContext));
423                table.setShowPagination(TagUtils.evaluateExpressionAsBoolean("showPagination", this.showPagination, this, pageContext));
424                table.setShowStatusBar(TagUtils.evaluateExpressionAsBoolean("showStatusBar", this.showStatusBar, this, pageContext));
425                table.setShowTitle(TagUtils.evaluateExpressionAsBoolean("showTitle", this.showTitle, this, pageContext));
426                table.setShowTooltips(TagUtils.evaluateExpressionAsBoolean("showTooltips", this.showTooltips, this, pageContext));
427                table.setSortRowsCallback(TagUtils.evaluateExpressionAsString("sortRowsCallback", this.sortRowsCallback, this, pageContext));
428                table.setSortable(TagUtils.evaluateExpressionAsBoolean("sortable", this.sortable, this, pageContext));
429                table.setState(TagUtils.evaluateExpressionAsString("state", this.state, this, pageContext));
430                table.setStateAttr(TagUtils.evaluateExpressionAsString("stateAttr", this.stateAttr, this, pageContext));
431                table.setStyle(TagUtils.evaluateExpressionAsString("style", style, this, pageContext));
432                table.setStyleClass(TagUtils.evaluateExpressionAsString("styleClass", this.styleClass, this, pageContext));
433                table.setTableId(TagUtils.evaluateExpressionAsString("tableId", tableId, this, pageContext));
434                table.setTheme(TagUtils.evaluateExpressionAsString("theme", this.theme, this, pageContext));
435                table.setTitle(TagUtils.evaluateExpressionAsString("title", this.title, this, pageContext));
436                table.setVar(TagUtils.evaluateExpressionAsString("var", this.var, this, pageContext));
437                table.setView(TagUtils.evaluateExpressionAsString("view", this.view, this, pageContext));
438                table.setWidth(TagUtils.evaluateExpressionAsString("width", this.width, this, pageContext));
439    
440                addTableAttributes(model, table);
441                model.addTable(table);
442            } catch (Exception e) {
443                throw new JspException("TableTag.doStartTag() Problem: " + ExceptionUtils.formatStackTrace(e));
444            }
445    
446            return EVAL_BODY_INCLUDE;
447        }
448    
449        /**
450         * Two things need to be accomplished here. First, need to iterate once over
451         * the columns to load up all the attributes. Second, need to iterate over
452         * the columns as many times as specified by the rowsDisplayed attribute so
453         * the columns row can be resolved. On each iteration over the columns the
454         * current bean in the collection is passed via the pageScope.
455         */
456        public int doAfterBody() throws JspException {
457            try {
458                if (iterator == null) {
459                    iterator = model.execute().iterator();
460                }
461    
462                if (iterator != null && iterator.hasNext()) {
463                    Object bean = iterator.next();
464                    model.setCurrentRowBean(bean);
465                    return EVAL_BODY_AGAIN;
466                }
467            } catch (Exception e) {
468                throw new JspException("TableTag.doAfterBody() Problem: " + ExceptionUtils.formatStackTrace(e));
469            }
470    
471            return SKIP_BODY;
472        }
473    
474        public int doEndTag() throws JspException {
475            try {
476                pageContext.getOut().println(model.getViewData());
477            } catch (Exception e) {
478                throw new JspException("TableTag.doEndTag() Problem: " + ExceptionUtils.formatStackTrace(e));
479            }
480    
481            return EVAL_PAGE;
482        }
