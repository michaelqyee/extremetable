016    package org.extremecomponents.table.cell;
017    
018    import org.apache.commons.lang.StringUtils;
019    import org.extremecomponents.table.bean.Column;
020    import org.extremecomponents.table.cell.Cell;
021    import org.extremecomponents.table.core.TableModel;
022    import org.extremecomponents.util.HtmlBuilder;
023    
024    /**
025     * @author Jeff Johnston
026     */
027    public class SelectAllHeaderCell implements Cell {
028        public String getExportDisplay(TableModel model, Column column) {
029            return column.getTitle();
030        }
031    
032        public String getHtmlDisplay(TableModel model, Column column) {
033            HtmlBuilder html = new HtmlBuilder();
034    
035            html.td(2);
036    
037            if (StringUtils.isNotEmpty(column.getHeaderClass())) {
038                html.styleClass(column.getHeaderClass());
039            }
040    
041            if (StringUtils.isNotEmpty(column.getHeaderStyle())) {
042                html.style(column.getHeaderStyle());
043            }
044    
045            if (StringUtils.isNotEmpty(column.getWidth())) {
046                html.width(column.getWidth());
047            }
048    
049            html.close();
050    
051            String controlName = column.getAlias() + "_selector";
052            String selectableControlName = column.getAlias();
053            
054            html.input("checkbox");
055            html.id(controlName);
056            html.name(controlName);
057            html.title("(Un)Select All");
058    
059            html.onclick("for(i = 0; i < document.getElementsByName('" + selectableControlName
060                    + "').length; i++)document.getElementsByName('" + selectableControlName
061                    + "').item(i).checked=document.getElementById('" + controlName + "').checked;");
062            html.close();
063    
064            html.tdEnd();
065    
066            return html.toString();
067        }
068    }
