067        public void beforeBody(TableModel model) {
068            div().styleClass("eXtremeTable").close();
069            toolbarPlacement(model);
070            tableStart(model);
071            statusBar(model);
072            filter(model);
073            header(model);
074            tbody(1).styleClass(TABLE_BODY).close();
075        }

077        public void body(TableModel model, Column column) {
078            if (column.isFirstColumn()) {
079                tr(1);
080                if (model.getRowHandler().isRowEven()) {
081                    rowStyleAndJavascript(model, BuilderConstants.ROW_EVEN_CSS);
082                } else {
083                    rowStyleAndJavascript(model, BuilderConstants.ROW_ODD_CSS);
084                }
085                close();
086            }
087    
088            append(column.getCellDisplay());
089    
090            if (column.isLastColumn()) {
091                trEnd(1);
092            }
093        }

095        public Object afterBody(TableModel model) {
096            totals(model);
097            tbodyEnd(1);
098            tableEnd(model);
099            newline().divEnd();
100    
101            return toString();
102        }

104        private void rowStyleAndJavascript(TableModel model, String defaultStyleClass) {
105            Row row = model.getRowHandler().getRow();
106    
107            String styleClass = row.getStyleClass();
108            String style = row.getStyle();
109    
110            if (StringUtils.isBlank(styleClass)) {
111                styleClass = defaultStyleClass;
112            }
113    
114            styleClass(styleClass);
115            style(style);
116            onclick(row.getOnclick());
117    
118            boolean highlightRow = row.isHighlightRow();
119            if (highlightRow) {
120                String highlightClass = row.getHighlightClass();
121                if (StringUtils.isNotBlank(row.getOnmouseover())) {
122                    onmouseover("this.className='" + highlightClass + "'; " + row.getOnmouseover());
123                } else {
124                    onmouseover("this.className='" + highlightClass + "'");
125                }
126    
127                if (StringUtils.isNotBlank(row.getOnmouseout())) {
128                    onmouseout("this.className='" + styleClass + "'; " + row.getOnmouseout());
129                } else {
130                    onmouseout("this.className='" + styleClass + "'");
131                }
132            } else {
133                onmouseover(row.getOnmouseover());
134                onmouseout(row.getOnmouseout());
135            }
136        }

138        public void toolbarPlacement(TableModel model) {
139            boolean showPagination = model.getTableHandler().getTable().isShowPagination();
140            boolean showExports = model.getExportHandler().showExports();
141    
142            if (!showPagination && !showExports && StringUtils.isBlank(model.getTableHandler().getTable().getTitle())) {
143                return;
144            }
145    
146            table(0).border("0").cellPadding("0").cellSpacing("0");
147            width(model.getTableHandler().getTable().getWidth()).close();
148    
149            tr(1).close();
150    
151            td(2).styleClass(TITLE).close();
152    
153            title(model);
154    
155            tdEnd();
156    
157            td(2).align("right").close();
158    
159            toolbar(model);
160    
161            tdEnd();
162    
163            trEnd(1);
164    
165            tableEnd(0);
166            newline();
167        }

169        public void title(TableModel model) {
170            String title = model.getTableHandler().getTable().getTitle();
171            if (StringUtils.isNotBlank(title)) {
172                span().close().append(title).spanEnd();
173            }
174        }

176        /**
177         * The pagination and export will display together.
178         */
179        public void toolbar(TableModel model) {
180            boolean showPagination = model.getTableHandler().getTable().isShowPagination();
181            boolean showExports = model.getExportHandler().showExports();
182    
183            if (!showPagination && !showExports) {
184                return;
185            }
186    
187            table(2).border("0").cellPadding("0").cellSpacing("1").styleClass(TOOLBAR).close();
188    
189            tr(3).close();
190    
191            toolbarFormStart(model);
192    
193            if (showPagination) {
194                toolbarPaginationIcons(model);
195                String separator = BuilderUtils.getImage(model, "separator");
196                td(4).rowSpan("2").styleClass(SEPARATOR).close().img(separator).tdEnd();
197    
198                rowsDisplayedDroplist(model);
199    
200                if (showExports) {
201                    td(4).rowSpan("2").styleClass(SEPARATOR).close().img(separator).tdEnd();
202                }
203            }
204    
205            if (showExports) {
206                toolbarExportIcons(model);
207            }
208    
209            trEnd(3);
210    
211            tr(3).close();
212    
213            formEnd();
214    
215            trEnd(3);
216    
217            tableEnd(2);
218            newline();
219            tabs(2);
220        }

226        private void toolbarFormStart(TableModel model) {
227            form();
228            name(model.getTableHandler().prefixWithTableId() + "toolbar");
229    
230            String action = model.getTableHandler().getTable().getAction();
231    
232            if (StringUtils.isNotEmpty(action)) {
233                action(action);

236            close();
237    
238            String hiddenFields = TreeRegistryUtils.getFormHiddenFields(model, true, true, false, false);

240            if (StringUtils.isNotEmpty(hiddenFields)) {
241                append(hiddenFields);
242            }
243        }

245        public void toolbarPaginationIcons(TableModel model) {
246            int page = model.getLimit().getPage();
247            int totalPages = BuilderUtils.getTotalPages(model);
248    
249            td(4).close();
250            if (!BuilderUtils.isFirstPageEnabled(page)) {
251                String firstPageImage = BuilderUtils.getImage(model, FIRST_PAGE_DISABLED);
252                img(firstPageImage);
253            } else {
254                String firstPageImage = BuilderUtils.getImage(model, FIRST_PAGE);
255                String firstPageTooltip = model.getMessages().getMessage(BuilderConstants.TOOLBAR_FIRST_PAGE_TOOLTIP);
256                paginationImage(model, 1, firstPageImage, firstPageTooltip);
257            }
258            tdEnd();
259    
260            // prev page
261            td(4).close();
262            if (!BuilderUtils.isPrevPageEnabled(page)) {
263                String prevPageImage = BuilderUtils.getImage(model, PREV_PAGE_DISABLED);
264                img(prevPageImage);
265            } else {
266                String prevPageImage = BuilderUtils.getImage(model, PREV_PAGE);
267                String prevPageTooltip = model.getMessages().getMessage(BuilderConstants.TOOLBAR_PREV_PAGE_TOOLTIP);
268                paginationImage(model, page - 1, prevPageImage, prevPageTooltip);
269            }
270            tdEnd();
271    
272            // next page
273            td(4).close();
274            if (!BuilderUtils.isNextPageEnabled(page, totalPages)) {
275                String nextPageImage = BuilderUtils.getImage(model, NEXT_PAGE_DISABLED);
276                img(nextPageImage);
277            } else {
278                String nextPageImage = BuilderUtils.getImage(model, NEXT_PAGE);
279                String nextPageTooltip = model.getMessages().getMessage(BuilderConstants.TOOLBAR_NEXT_PAGE_TOOLTIP);
280                paginationImage(model, page + 1, nextPageImage, nextPageTooltip);
281            }
282            tdEnd();
283    
284            td(4).close();
285            if (!BuilderUtils.isLastPageEnabled(page, totalPages)) {
286                String lastPageImage = BuilderUtils.getImage(model, LAST_PAGE_DISABLED);
287                img(lastPageImage);
288            } else {
289                String lastPageImage = BuilderUtils.getImage(model, LAST_PAGE);
290                String lastPageTooltip = model.getMessages().getMessage(BuilderConstants.TOOLBAR_LAST_PAGE_TOOLTIP);
291                paginationImage(model, totalPages, lastPageImage, lastPageTooltip);
292            }
293            tdEnd();
294        }

296        public void paginationImage(TableModel model, int page, String image, String tooltip) {
297            a();
298            quote();
299    
300            String action = model.getTableHandler().getTable().getAction();
301    
302            if (StringUtils.isNotEmpty(action)) {
303                append(action);
304            }
305    
306            question().append(model.getTableHandler().prefixWithTableId()).append(TableConstants.PAGE).equals().append("" + page);
307            append(TreeRegistryUtils.getURLParameterString(model, true, true, false, true));
308            quote().close();
309            img(image, tooltip);
310            aEnd();

322        public void rowsDisplayedDroplist(TableModel model) {
323            int rowsDisplayed = model.getTableHandler().getTable().getRowsDisplayed();
324            int medianRowsDisplayed = model.getTableHandler().getTable().getMedianRowsDisplayed();
325            int maxRowsDisplayed = model.getTableHandler().getTable().getMaxRowsDisplayed();
326            int currentRowsDisplayed = model.getLimit().getCurrentRowsDisplayed();
327    
328            td(4).width("20").close();
329            newline();
330            tabs(4);
331            select().name(model.getTableHandler().prefixWithTableId() + TableConstants.CURRENT_ROWS_DISPLAYED);
332    
333            StringBuffer onchange = new StringBuffer();
334            onchange.append(HtmlViewUtils.paginationJavaScript(model));
335            onchange(onchange.toString());
336    
337            close();
338    
339            newline();
340            tabs(4);
341    
342            // default rows
343            option().value(String.valueOf(rowsDisplayed));
344            if (currentRowsDisplayed == rowsDisplayed) {
345                append(" selected");
346            }
347            close();
348            append(String.valueOf(rowsDisplayed));
349            optionEnd();
350    
351            // median rows
352            option().value(String.valueOf(medianRowsDisplayed));
353            if (currentRowsDisplayed == medianRowsDisplayed) {
354                append(" selected");
355            }
356            close();
357            append(String.valueOf(medianRowsDisplayed));
358            optionEnd();
359    
360            // max rows
361            option().value(String.valueOf(maxRowsDisplayed));
362            if (currentRowsDisplayed == maxRowsDisplayed) {
363                append(" selected");
364            }
365            close();
366            append(String.valueOf(maxRowsDisplayed));
367            optionEnd();
368    
369            newline();
370            tabs(4);
371            selectEnd();
372    
373            img(BuilderUtils.getImage(model, "rowsDisplayed"));
374    
375            tdEnd();
376        }

378        private void exportImage(TableModel model, Export export) {
379            a();
380            quote();
381    
382            String action = model.getTableHandler().getTable().getAction();
383    
384            if (StringUtils.isNotEmpty(action)) {
385                append(action);
386            }
387    
388            question().append(TableConstants.EXPORT_TABLE_ID).equals().append(model.getTableHandler().getTable().getTableId()).ampersand().append(
389                    model.getTableHandler().prefixWithTableId() + TableConstants.EXPORT_VIEW).equals().append(export.getView()).ampersand().append(
390                    model.getTableHandler().prefixWithTableId() + TableConstants.EXPORT_FILE_NAME).equals().append(export.getFileName());
391            append(TreeRegistryUtils.getURLParameterString(model, true, true, true, true));
392            quote();
393            close();
394    
395            String imageName = BuilderUtils.getImage(model, export.getImageName());
396    
397            if (StringUtils.isNotEmpty(imageName)) {
398                String tooltip = export.getTooltip();
399                if (StringUtils.isEmpty(tooltip)) {
400                    img(imageName);
401                } else {
402                    img(imageName, tooltip);
403                }
404            } else {
405                nbsp().append(export.getView()).nbsp();
406            }
407    
408            aEnd();
409        }

411        /**
412         * The HTML for the filter search and clear buttons.
413         */
414        public void statusBar(TableModel model) {
415            if (!model.getTableHandler().getTable().isShowStatusBar() && !model.getTableHandler().getTable().isFilterable()) {
416                return;
417            }
418    
419            tr(1).close();
420    
421            td(2).colSpan(String.valueOf(model.getColumnHandler().columnCount())).close();
422    
423            table(2).border("0").cellPadding("0").cellSpacing("0").width("100%").close();
424            tr(3).close();
425    
426            if (model.getTableHandler().getTable().isShowStatusBar()) {
427                td(4).styleClass(STATUS_BAR).close();
428    
429                if (model.getLimit().getTotalRows() == 0) {
430                    append(model.getMessages().getMessage(BuilderConstants.STATUSBAR_NO_RESULTS_FOUND));
431                } else {
432                    Integer total = new Integer(model.getLimit().getTotalRows());
433                    Integer from = new Integer(model.getLimit().getRowStart() + 1);
434                    Integer to = new Integer(model.getLimit().getRowEnd());
435                    Object[] messageArguments = { total, from, to };
436                    append(model.getMessages().getMessage(BuilderConstants.STATUSBAR_RESULTS_FOUND, messageArguments));
437                }
438    
439                tdEnd();
440            }
441    
442            if (model.getTableHandler().getTable().isFilterable()) {
443                td(4).styleClass(FILTER_BUTTONS).close();
444    
445                String imageSearchArrow = BuilderUtils.getImage(model, SEARCH_ARROW_IMAGE);
446    
447                img(imageSearchArrow);
448    
449                nbsp();
450    
451                a().quote().append(HtmlViewUtils.filterJavaScript(model)).quote().close();
452    
453                String imageSearch = BuilderUtils.getImage(model, SEARCH_IMAGE);
454    
455                if (StringUtils.isNotEmpty(imageSearch)) {
456                    String searchTooltip = model.getMessages().getMessage(BuilderConstants.TOOLBAR_FILTER_TOOLTIP);
457                    img(imageSearch, searchTooltip);
458                } else {
459                    append(" Filter ");
460                }
461    
462                aEnd();
463    
464                nbsp();
465    
466                a().quote().append(
467                        "javascript:document.forms." + model.getTableHandler().prefixWithTableId() + "filter." + model.getTableHandler().prefixWithTableId() + TableConstants.FILTER
468                                + TableConstants.ACTION + ".value='" + TableConstants.CLEAR_ACTION + "';document.forms." + model.getTableHandler().prefixWithTableId() + "filter.submit()").quote().close();
469    
470                String imageClear = BuilderUtils.getImage(model, CLEAR_IMAGE);
471    
472                if (StringUtils.isNotEmpty(imageClear)) {
473                    String clearTooltip = model.getMessages().getMessage(BuilderConstants.TOOLBAR_CLEAR_TOOLTIP);
474                    img(imageClear, clearTooltip);
475                } else {
476                    append(" Clear ");
477                }
478    
479                aEnd();
480    
481                tdEnd();
482            }
483    
484            trEnd(3);
485            tableEnd(2);
486            newline();
487            tabs(2);
488    
489            tdEnd();
490            trEnd(1);
491            tabs(2);
492            newline();
493        }

535        /**
536         * The filter row of the table. Typically sits below the header (if
537         * enabled). Will contain input or select type fields to allow the user to
538         * filter the result set.
539         */
540        public void filter(TableModel model) {
541            if (!model.getTableHandler().getTable().isFilterable()) {
542                return;
543            }
544    
545            filterFormStart(model);
546    
547            tr(1).styleClass(FILTER).id(FILTER).close();
548    
549            List columns = model.getColumnHandler().getFilterColumns();
550            for (Iterator iter = columns.iterator(); iter.hasNext();) {
551                Column column = (Column) iter.next();
552                append(column.getCellDisplay());
553            }
554    
555            trEnd(1);
556            formEnd();
557        }

583        public void totals(TableModel model) {
584            Column calcColumn = model.getColumnHandler().getFirstCalcColumn();
585            if (calcColumn == null) {
586                return;
587            }
588    
589            String calcTitle[] = calcColumn.getCalcTitle();
590            if (calcTitle != null && calcTitle.length > 0) {
591                tr(1).close();
592                td(2).styleClass(TABLE_TOTAL_TITLE).colSpan("" + model.getColumnHandler().columnCount()).close();
593                for (int i = 0; i < calcTitle.length; i++) {
594                    String title = calcTitle[i];
595                    append(title);
596                }
597                tdEnd();
598                trEnd(1);
599            }
600    
601            tr(1).close();
602            for (Iterator iter = model.getColumnHandler().getColumns().iterator(); iter.hasNext();) {
603                Column column = (Column) iter.next();
604                if (column.isCalculated()) {
605                    td(2).styleClass(TABLE_TOTALS).close();
606                    CalcResult calcValues[] = model.getColumnHandler().getCalcResults(column);
607                    for (int i = 0; i < calcValues.length; i++) {
608                        CalcResult calcValue = calcValues[i];
609                        Number value = calcValue.getValue();
610                        append(ExtremeUtils.formatNumber(column.getFormat(), value, model.getLocale()));
611                        if (calcValues.length > 0 && i + 1 != calcValues.length) {
612                            append(" / ");
613                        }
614                    }
615                } else {
616                    td(2).styleClass(TABLE_TOTALS_EMPTY).close();
617                    nbsp();
618                }
619    
620                tdEnd();
621            }
622            trEnd(1);
623        }
